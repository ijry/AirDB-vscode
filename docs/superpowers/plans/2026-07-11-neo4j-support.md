# Neo4j Support Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Neo4j as a first-class NoSQL backend with connection setup, tree browsing, and Cypher execution through the existing result surface.

**Architecture:** Neo4j is a NoSQL backend stored under `CacheKey.NOSQL_CONNECTION`, routed through a dedicated `Neo4jConnection`, and displayed with dedicated `src/model/neo4j/*` tree nodes. It does not use SQL dialects or relational schema/table nodes.

**Tech Stack:** TypeScript, VS Code extension APIs, Vue 2 connection page, official `neo4j-driver` package, mocked Node tests with `ts-node/register/transpile-only`.

## Global Constraints

- Use `neo4j-driver` dependency range `^6.2.0`.
- Default Neo4j host is `127.0.0.1`.
- Default Neo4j port is `7687`.
- Default Neo4j user is `neo4j`.
- Default Neo4j password is an empty string.
- Default Neo4j database is `neo4j`.
- Default `useSSL` is `false`.
- Default `connectTimeout` is `5000`.
- Default `requestTimeout` is `10000`.
- Use `bolt://host:port` without SSL and `bolt+s://host:port` with SSL.
- `connectionUrl` overrides the generated URI.
- Metadata commands are `SHOW DATABASES YIELD name RETURN name ORDER BY name`, `CALL db.labels() YIELD label RETURN label ORDER BY label`, and `CALL db.relationshipTypes() YIELD relationshipType RETURN relationshipType ORDER BY relationshipType`.
- Do not add graph canvas visualization, node/relationship editing, APOC workflows, admin panels, subscriptions, or live integration tests in this version.
- Preserve unrelated untracked files: `docs/superpowers/plans/2026-07-11-doris-support.md` and `logs/`.

const assert = require("assert");
const { requireTs } = require("./testSetup");
const { Neo4jConnection, serializeNeo4jValue } = requireTs("src/service/connect/neo4jConnection.ts");

function record(values) {
  return { keys: Object.keys(values), get: (key) => values[key], toObject: () => values };
}

const calls = [];
const fakeDriver = {
  verifyConnectivity: async () => calls.push(["verifyConnectivity"]),
  session: (options) => ({
    run: async (cypher) => {
      calls.push(["run", options, cypher]);
      if (cypher.startsWith("SHOW DATABASES")) return { records: [record({ name: "neo4j" }), record({ name: "system" })] };
      if (cypher.includes("db.labels")) return { records: [record({ label: "Person" })] };
      if (cypher.includes("db.relationshipTypes")) return { records: [record({ relationshipType: "ACTED_IN" })] };
      return { records: [record({ name: "Keanu", born: { low: 1964, high: 0, toNumber: () => 1964, inSafeRange: () => true } })] };
    },
    close: async () => calls.push(["sessionClose"]),
  }),
  close: async () => calls.push(["driverClose"]),
};

const connection = new Neo4jConnection({ database: "movies" }, () => fakeDriver);
await new Promise((resolve, reject) => connection.connect((err) => err ? reject(err) : resolve()));
assert.strictEqual(connection.isAlive(), true);

await new Promise((resolve, reject) => {
  connection.query("MATCH (n) RETURN n.name AS name", (err, rows, fields, total) => {
    if (err) return reject(err);
    assert.deepStrictEqual(rows, [{ name: "Keanu", born: 1964 }]);
    assert.deepStrictEqual(fields.map((field) => field.name), ["name", "born"]);
    assert.strictEqual(total, 1);
    resolve();
  });
});

assert.deepStrictEqual(await connection.listDatabases(), ["neo4j", "system"]);
assert.deepStrictEqual(await connection.listLabels(), ["Person"]);
assert.deepStrictEqual(await connection.listRelationshipTypes(), ["ACTED_IN"]);

assert.deepStrictEqual(serializeNeo4jValue({ identity: { toString: () => "1" }, labels: ["Person"], properties: { age: { toNumber: () => 42, inSafeRange: () => true } }, elementId: "1" }), {
  identity: "1",
  labels: ["Person"],
  properties: { age: 42 },
  elementId: "1",
});

connection.end();
assert.strictEqual(connection.isAlive(), false);
assert.ok(calls.some((call) => call[0] === "driverClose"));
console.log("neo4jConnection tests passed");
